type QuestionSource = "resume" | "job" | "behavioral";

type GeneratedQuestion = {
  question_text: string;
  source_type: QuestionSource;
  reference_context: Record<string, unknown>;
  mapped_skill?: string;
  phase_hint?: "warmup" | "core" | "probe" | "closing";
  difficulty_level?: number;
  allow_follow_up?: boolean;
};

type ResumeContext = {
  rawResume: string | null;
  skills: string[];
  claims: string[];
  experienceYears: number | null;
};

type JobContext = {
  title: string | null;
  description: string | null;
  coreSkills: string[];
  responsibilities: string[];
  difficultyProfile: string | null;
};

type InterviewContext = {
  roleTitle: string | null;
  resume: ResumeContext;
  job: JobContext;
  interviewSkills: string[];
  overlapSkills: string[];
};

type Distribution = Record<QuestionSource, number>;

const GENERIC_PATTERNS = [
  /tell me about yourself/i,
  /tell me about your experience/i,
  /walk me through your background/i,
  /describe your background/i,
  /what do you do/i,
  /tell me about a recent problem/i,
];
const WEAK_LEAD_IN_PATTERNS = [
  /^describe a situation\b/i,
  /^share an example\b/i,
  /^tell me about a time\b/i,
  /^can you walk me through a situation\b/i,
];

const INTENT_PATTERNS: Array<{ intent: string; pattern: RegExp }> = [
  { intent: "optimize", pattern: /\b(optimi[sz]e|performance|slow query|bottleneck)\b/i },
  { intent: "debug", pattern: /\b(debug|troubleshoot|incident|issue|failure|outage)\b/i },
  { intent: "design", pattern: /\b(design|architect|structure|approach|plan)\b/i },
  { intent: "migration", pattern: /\b(migration|migrate|upgrade|moderni[sz]e)\b/i },
  { intent: "ownership", pattern: /\b(led|owned|managed|responsible|stakeholder)\b/i },
  { intent: "behavior", pattern: /\b(conflict|decision|feedback|trade-?off|priority|pressure)\b/i },
];

function normalizePositiveInteger(value: number | null | undefined) {
  if (typeof value !== "number" || !Number.isFinite(value)) {
    return 0;
  }

  const normalized = Math.floor(value);
  return normalized > 0 ? normalized : 0;
}

function normalizeText(value: string | null | undefined) {
  return value?.replace(/\s+/g, " ").trim() ?? "";
}

const ROLE_SEGMENT_SPLIT = /\s+\|\s+|\s+-\s+|\s+–\s+|\s+—\s+|,\s*/;
const ROLE_KEYWORDS = [

  "analyst",

];

  "bangalore",

]);
const GENDERED_TERMS = /\b(he|she|him|his|her|hers)\b/gi;
const ROLE_ABBREVIATIONS: Array<[RegExp, string]> = [
  [/\bsr\.?\b/gi, "Senior"],
  [/\bjr\.?\b/gi, "Junior"],
  [/\bassoc\.?\b/gi, "Associate"],
];

function isRoleNoiseToken(token: string) {
  return ROLE_NOISE_TOKENS.has(token.toLowerCase());
}

function standardizeRoleCasing(value: string) {
  return value
    .split(/\s+/)
    .map((token) => {
      if (/^[A-Z0-9]{2,}$/.test(token)) {
        return token;
      }

      const lower = token.toLowerCase();
      return lower.charAt(0).toUpperCase() + lower.slice(1);
    })
    .join(" ");
}

function cleanAnchorText(value: string | null | undefined) {
  let cleaned = normalizeText(value);

  if (!cleaned) {
    return null;
  }

  cleaned = cleaned
    .replace(/[|_/]+/g, " ")
    .replace(/[()[\]{}]+/g, " ")
    .replace(GENDERED_TERMS, " ")
    .replace(
      /\b(remote|hybrid|onsite|on-site|work from home|wfh|delhi|bangalore|bengaluru|blr|chennai|hyderabad|kolkata|mumbai|pune|noida|gurgaon|gurugram|dubai|singapore|holland|netherlands|india|uk|usa|us)\b/gi,
      " "
    )
    .replace(
      /\b(day|night|rotational|general|morning|evening|first|second|third|1st|2nd|3rd|ist)\s+shift\b/gi,
      " "
    )
    .replace(/\bshift\s*[a-z0-9:+-]*\b/gi, " ")
    .replace(/\s+/g, " ")
    .replace(/^[,;:.\-]+|[,;:.\-]+$/g, "")
    .trim();

  return cleaned || null;
}

function containsLocationNoise(value: string | null | undefined) {
  const normalized = normalizeText(value);

  if (!normalized) {
    return false;
  }

  return /\b(delhi|bangalore|bengaluru|blr|chennai|hyderabad|kolkata|mumbai|pune|noida|gurgaon|gurugram|dubai|singapore|holland|netherlands|india|uk|usa|us|remote|hybrid|onsite|on-site|wfh)\b/i.test(
    normalized
  );
}

function containsGenderAssumption(value: string | null | undefined) {
  return /\b(he|she|him|his|her|hers)\b/i.test(normalizeText(value));
}

function finalizeQuestionText(value: string | null | undefined) {
  let cleaned = normalizeText(value);

  if (!cleaned) {
    return null;
  }

  cleaned = cleaned
    .replace(/\b(he|she|him)\b/gi, "the candidate")
    .replace(/\b(his|her|hers)\b/gi, "the candidate's")
    .replace(/\s+/g, " ")
    .replace(/\s+([?!.,;:])/g, "$1")
    .trim();

  if (!/[?.!]$/.test(cleaned)) {
    cleaned = `${cleaned}?`;
  }

  return cleaned.charAt(0).toUpperCase() + cleaned.slice(1);
}

function isReadableQuestion(value: string | null | undefined) {
  const normalized = normalizeText(value);

  if (!normalized) {
    return false;
  }

  const words = normalized.split(/\s+/);

  return (
    words.length >= 6 &&
    words.length <= 35 &&
    /^[A-Z]/.test(normalized) &&
    /[?.!]$/.test(normalized) &&
    !/\b(\w+)\s+\1\b/i.test(normalized) &&
    !/[|_]{2,}/.test(normalized)
  );
}

function scoreRoleSegment(segment: string) {
  const tokens = normalizeText(segment).toLowerCase().split(/\s+/).filter(Boolean);

  if (tokens.length === 0) {
    return Number.NEGATIVE_INFINITY;
  }

  return tokens.reduce((score, token) => {
    if (ROLE_KEYWORDS.some((keyword) => token.includes(keyword))) {
      return score + 3;
    }

    if (isRoleNoiseToken(token)) {
      return score - 3;
    }

    if (/shift|remote|hybrid|onsite|on-site/.test(token)) {
      return score - 2;
    }

    return score + 1;
  }, 0);
}

function cleanRoleTitle(value: string | null | undefined) {
  let cleaned = normalizeText(value);

  if (!cleaned) {
    return null;
  }

  cleaned = cleaned.replace(
    /\((?:[^)]*\b(?:remote|hybrid|onsite|on-site|wfh|work from home|shift|bangalore|bengaluru|holland|netherlands|india|uk|usa|us)\b[^)]*)\)/gi,
    " "
  );

  for (const [pattern, replacement] of ROLE_ABBREVIATIONS) {
    cleaned = cleaned.replace(pattern, replacement);
  }

  const segments = cleaned
    .split(ROLE_SEGMENT_SPLIT)
    .map((segment) => normalizeText(segment))
    .filter(Boolean);

  if (segments.length > 1) {
    cleaned = [...segments].sort((left, right) => scoreRoleSegment(right) - scoreRoleSegment(left))[0];
  }

  cleaned = cleaned
    .replace(/\b(remote|hybrid|onsite|on-site|work from home|wfh)\b/gi, " ")
    .replace(
      /\b(day|night|rotational|general|morning|evening|first|second|third|1st|2nd|3rd|us|uk|europe|emea|apac|ist)\s+shift\b/gi,
      " "
    )
    .replace(/\bshift\s*[a-z0-9:+-]*\b/gi, " ")
    .replace(/\s+/g, " ")
    .trim();

  let tokens = cleaned.split(/\s+/).filter(Boolean);

  while (tokens.length > 0 && isRoleNoiseToken(tokens[0])) {
    tokens = tokens.slice(1);
  }

  while (tokens.length > 0 && isRoleNoiseToken(tokens[tokens.length - 1])) {
    tokens = tokens.slice(0, -1);
  }

  cleaned = standardizeRoleCasing(tokens.join(" "));
  return cleaned || null;
}

function normalizeQuestionText(value: string | null | undefined) {
  return normalizeText(value)
    .toLowerCase()
    .replace(/[^a-z0-9\s]/g, " ")
    .replace(/\s+/g, " ")
    .trim();
}

function tokenizeQuestion(value: string | null | undefined) {
  return normalizeQuestionText(value)
    .split(" ")
    .filter((token) => token.length > 2 && !STOP_WORDS.has(token));
}

function uniqueStrings(values: Array<string | null | undefined>) {
  return [...new Set(values.map((value) => normalizeText(value)).filter(Boolean))];
}

function normalizeAnchorToken(value: string | null | undefined) {
  return normalizeText(value)
    .toLowerCase()
    .replace(/[^a-z0-9\s]/g, " ")
    .replace(/\s+/g, " ")
    .trim();
}

function extractResponsibilityAnchors(value: string | null | undefined, limit = 6) {
  const normalized = normalizeText(value);

  if (!normalized) {
    return [];
  }

  return uniqueStrings(
    normalized
      .split(/[.;\n\r]+/)
      .map((segment) => cleanAnchorText(segment))
      .filter((segment) => {
        if (!segment) {
          return false;
        }
        const wordCount = segment.split(/\s+/).length;
        return (
          wordCount >= 4 &&
          wordCount <= 18 &&
          /\b(handle|manage|maintain|monitor|optimi[sz]e|troubleshoot|support|design|deliver|migrate|automate|secure|improve|own|lead)\b/i.test(
            segment
          )
        );
      })
      .map((segment) => segment.replace(/^(responsibilities?|requirements?)\s*[:\-]\s*/i, ""))
  ).slice(0, limit);
}

function extractClaimAnchors(values: Array<string | null | undefined>, limit = 6) {
  const result: string[] = [];

  for (const value of values) {
    const normalized = normalizeText(value);
    if (!normalized) {
      continue;
    }

    const segments = normalized.split(/[.;,\n\r]+/);
    for (const segment of segments) {
      const claim = cleanAnchorText(segment);
      if (!claim) {
        continue;
      }
      const wordCount = claim.split(/\s+/).length;
      if (claim && wordCount >= 2 && wordCount <= 10 && !result.includes(claim)) {
        result.push(claim);
      }

      if (result.length >= limit) {
        return result;
      }
    }
  }

  return result;
}

function computeSkillOverlap(params: {
  resumeSkills: string[];
  resumeClaims: string[];
  jobSkills: string[];
}) {
  const resumeAnchors = uniqueStrings([...params.resumeSkills, ...extractClaimAnchors(params.resumeClaims)]);
  const jobSkills = uniqueStrings(params.jobSkills);
  const jobSkillMap = new Map(jobSkills.map((skill) => [normalizeAnchorToken(skill), skill]));
  const overlap: string[] = [];

  for (const anchor of resumeAnchors) {
    const normalized = normalizeAnchorToken(anchor);
    const direct = jobSkillMap.get(normalized);

    if (direct && !overlap.includes(direct)) {
      overlap.push(direct);
      continue;
    }

    const partial = jobSkills.find((jobSkill) => {
      const normalizedJobSkill = normalizeAnchorToken(jobSkill);
      return (
        normalized.length >= 4 &&
        normalizedJobSkill.length >= 4 &&
        (normalized.includes(normalizedJobSkill) || normalizedJobSkill.includes(normalized))
      );
    });

    if (partial && !overlap.includes(partial)) {
      overlap.push(partial);
    }
  }

  return overlap;
}

function pickQuestionAnchor(params: {
  sourceType: QuestionSource;
  overlapSkills: string[];
  jobSkills: string[];
  responsibilities: string[];
  resumeSkills: string[];
  resumeClaims: string[];
  roleTitle: string | null;
  index: number;
}) {
  const overlapSkill =
    cleanAnchorText(
      params.overlapSkills[params.index % Math.max(params.overlapSkills.length, 1)] ?? null
    );
  const jobSkill =
    cleanAnchorText(params.jobSkills[params.index % Math.max(params.jobSkills.length, 1)] ?? null);
  const responsibility =
    cleanAnchorText(
      params.responsibilities[params.index % Math.max(params.responsibilities.length, 1)] ?? null
    );
  const resumeSkill =
    cleanAnchorText(
      params.resumeSkills[params.index % Math.max(params.resumeSkills.length, 1)] ?? null
    );
  const claimAnchors = extractClaimAnchors(params.resumeClaims);
  const resumeClaim =
    claimAnchors[params.index % Math.max(claimAnchors.length, 1)] ?? null;

  if (overlapSkill) {
    return overlapSkill;
  }

  if (jobSkill) {
    return jobSkill;
  }

  if (params.sourceType === "resume") {
    return resumeSkill ?? resumeClaim ?? responsibility ?? params.roleTitle ?? "your recent experience";
  }

  if (params.sourceType === "behavioral") {
    return responsibility ?? jobSkill ?? resumeSkill ?? params.roleTitle ?? "your work";
  }

  return jobSkill ?? responsibility ?? resumeSkill ?? resumeClaim ?? params.roleTitle ?? "the role requirements";
}

function deriveQuestionTargetFromDuration(durationMinutes: number | null | undefined) {
  const duration = normalizePositiveInteger(durationMinutes);

  if (duration >= 60) return 11;
  if (duration >= 45) return 8;
  if (duration >= 30) return 6;
  if (duration >= 20) return 4;
  if (duration >= 15) return 3;
  if (duration >= 10) return 2;
  if (duration > 0) return 2;
  return 6;
}

function resolveEffectiveQuestionCount(params: {
  configuredCount?: number | null;
  durationMinutes?: number | null;
}) {
  const configuredCount = normalizePositiveInteger(params.configuredCount);
  const durationMinutes = normalizePositiveInteger(params.durationMinutes);
  const durationTarget = deriveQuestionTargetFromDuration(durationMinutes);
  const durationFloor =
    durationMinutes >= 60 ? 8 :
    durationMinutes >= 45 ? 6 :
    durationMinutes >= 30 ? 5 :
    durationMinutes >= 20 ? 3 :
    durationMinutes >= 15 ? 2 :
    durationMinutes >= 10 ? 2 :
    durationMinutes > 0 ? 1 : 1;

  if (durationMinutes === 0) {
    return Math.max(configuredCount, 1);
  }

  if (configuredCount === 0) {
    return durationTarget;
  }

  return Math.min(Math.max(configuredCount, durationFloor), durationTarget);
}

function buildDistribution(totalQuestions: number, hasResumeContext: boolean): Distribution {
  const weights = hasResumeContext
    ? { resume: 0.3, job: 0.5, behavioral: 0.2 }
    : { resume: 0, job: 0.7, behavioral: 0.3 };

  const base = {
    resume: Math.floor(totalQuestions * weights.resume),
    job: Math.floor(totalQuestions * weights.job),
    behavioral: Math.floor(totalQuestions * weights.behavioral),
  } satisfies Distribution;

  let assigned = base.resume + base.job + base.behavioral;
  const order: QuestionSource[] = hasResumeContext
    ? ["job", "resume", "behavioral"]
    : ["job", "behavioral", "resume"];

  while (assigned < totalQuestions) {
    const next = order[(assigned - (base.resume + base.job + base.behavioral)) % order.length];
    base[next] += 1;
    assigned += 1;
  }

  if (hasResumeContext && base.resume === 0 && totalQuestions >= 3) {
    base.resume = 1;
    if (base.job > base.behavioral) {
      base.job -= 1;
    } else {
      base.behavioral -= 1;
    }
  }

  return base;
}

function extractClaimPhrases(value: unknown): string[] {
  const phrases: string[] = [];

  const walk = (input: unknown) => {
    if (!input) {
      return;
    }

    if (typeof input === "string") {
      const normalized = normalizeText(input);
      if (normalized && normalized.length >= 8) {
        phrases.push(normalized);
      }
      return;
    }

    if (Array.isArray(input)) {
      input.forEach(walk);
      return;
    }

    if (typeof input === "object") {
      Object.values(input as Record<string, unknown>).forEach(walk);
    }
  };

  walk(value);

  return uniqueStrings(phrases).slice(0, 12);
}

function detectIntent(questionText: string) {
  for (const candidate of INTENT_PATTERNS) {
    if (candidate.pattern.test(questionText)) {
      return candidate.intent;
    }
  }

  return "general";
}

function similarityScore(left: string, right: string) {
  const leftTokens = new Set(tokenizeQuestion(left));
  const rightTokens = new Set(tokenizeQuestion(right));

  if (leftTokens.size === 0 || rightTokens.size === 0) {
    return 0;
  }

  let overlap = 0;
  for (const token of leftTokens) {
    if (rightTokens.has(token)) {
      overlap += 1;
    }
  }

  const unionSize = new Set([...leftTokens, ...rightTokens]).size;
  return unionSize === 0 ? 0 : overlap / unionSize;
}

function isGenericQuestion(questionText: string) {
  return GENERIC_PATTERNS.some((pattern) => pattern.test(questionText));
}

function hasWeakLeadIn(questionText: string) {
  return WEAK_LEAD_IN_PATTERNS.some((pattern) => pattern.test(normalizeText(questionText)));
}

function containsAnchor(questionText: string, anchors: string[]) {
  const normalized = normalizeQuestionText(questionText);
  return anchors.some((anchor) => {
    const normalizedAnchor = normalizeQuestionText(anchor);
    return normalizedAnchor.length >= 3 && normalized.includes(normalizedAnchor);
  });
}

function buildContextAnchors(context: InterviewContext) {
  const resumeAnchors = uniqueStrings([
    ...context.overlapSkills,
    ...context.resume.skills,
    ...context.resume.claims,
    context.resume.experienceYears ? `${context.resume.experienceYears} years` : null,
  ]);
  const jobAnchors = uniqueStrings([
    ...context.overlapSkills,
    ...context.job.coreSkills,
    ...context.job.responsibilities,
    ...context.interviewSkills,
    context.roleTitle,
  ]);
  const behavioralAnchors = uniqueStrings([
    ...context.overlapSkills,
    ...context.job.responsibilities,
    ...context.job.coreSkills,
    ...context.interviewSkills,
    context.roleTitle,
  ]);

  return { resumeAnchors, jobAnchors, behavioralAnchors };
}

function buildAllowedSkillAnchors(context: InterviewContext) {
  return uniqueStrings([
    ...context.overlapSkills,
    ...context.job.coreSkills,
    ...context.interviewSkills,
    ...context.resume.skills,
    ...extractClaimAnchors(context.resume.claims),
  ]);
}

function resolveMappedSkill(question: GeneratedQuestion, context: InterviewContext) {
  const allowedSkillAnchors = buildAllowedSkillAnchors(context);
  const candidateAnchors = uniqueStrings([
    question.mapped_skill ?? null,
    typeof question.reference_context.anchor === "string"
      ? question.reference_context.anchor
      : null,
    ...allowedSkillAnchors.filter((anchor) =>
      normalizeQuestionText(question.question_text).includes(normalizeQuestionText(anchor))
    ),
  ]);

  const exactMatch = candidateAnchors.find((candidate) =>
    allowedSkillAnchors.some(
      (allowed) => normalizeAnchorToken(allowed) === normalizeAnchorToken(candidate)
    )
  );

  if (exactMatch) {
    return exactMatch;
  }

  const partialMatch = candidateAnchors.find((candidate) =>
    allowedSkillAnchors.some((allowed) => {
      const normalizedAllowed = normalizeAnchorToken(allowed);
      const normalizedCandidate = normalizeAnchorToken(candidate);

      return (
        normalizedAllowed.length >= 4 &&
        normalizedCandidate.length >= 4 &&
        (normalizedAllowed.includes(normalizedCandidate) ||
          normalizedCandidate.includes(normalizedAllowed))
      );
    })
  );

  if (!partialMatch) {
    return null;
  }

  return (
    allowedSkillAnchors.find((allowed) => {
      const normalizedAllowed = normalizeAnchorToken(allowed);
      const normalizedCandidate = normalizeAnchorToken(partialMatch);
      return (
        normalizedAllowed.length >= 4 &&
        normalizedCandidate.length >= 4 &&
        (normalizedAllowed.includes(normalizedCandidate) ||
          normalizedCandidate.includes(normalizedAllowed))
      );
    }) ?? null
  );
}

function isSkillDrivenQuestion(question: GeneratedQuestion, context: InterviewContext) {
  const mappedSkill = resolveMappedSkill(question, context);
  return Boolean(mappedSkill) && containsAnchor(question.question_text, [mappedSkill]);
}

function getQuestionLeadKey(questionText: string) {
  return tokenizeQuestion(questionText).slice(0, 3).join(" ");
}

function isContextualQuestion(question: GeneratedQuestion, context: InterviewContext) {
  const anchors = buildContextAnchors(context);

  if (question.source_type === "resume") {
    return containsAnchor(question.question_text, anchors.resumeAnchors);
  }

  if (question.source_type === "job") {
    return containsAnchor(question.question_text, anchors.jobAnchors);
  }

  return containsAnchor(question.question_text, anchors.behavioralAnchors);
}

function normalizeQuestions(
  payload: Record<string, unknown>,
  context: InterviewContext
): GeneratedQuestion[] {
  const rawQuestions = Array.isArray(payload.questions) ? payload.questions : [];

  return rawQuestions
    .map((item) => {
      if (!item || typeof item !== "object") {
        return null;
      }

      const record = item as Record<string, unknown>;
      const questionText =
        typeof record.question_text === "string" ? record.question_text.trim() : "";
      const sourceType =
        record.source_type === "resume" ||
        record.source_type === "job" ||
        record.source_type === "behavioral"
          ? record.source_type
          : null;
      const mappedSkill =
        typeof record.mapped_skill === "string" ? record.mapped_skill.trim() : "";
      const phaseHint =
        record.phase_hint === "warmup" ||
        record.phase_hint === "core" ||
        record.phase_hint === "probe" ||
        record.phase_hint === "closing"
          ? record.phase_hint
          : "core";
      const difficultyLevel =
        typeof record.difficulty_level === "number"
          ? Math.max(1, Math.min(5, Math.round(record.difficulty_level)))
          : 3;

      const cleanedQuestionText = finalizeQuestionText(questionText);

      if (!cleanedQuestionText || !sourceType) {
        return null;
      }

      const normalizedQuestion = {
        question_text: cleanedQuestionText,
        source_type: sourceType,
        reference_context:
          record.reference_context && typeof record.reference_context === "object"
            ? (record.reference_context as Record<string, unknown>)
            : {},
        mapped_skill: cleanAnchorText(mappedSkill) || undefined,
        phase_hint: phaseHint,
        difficulty_level: difficultyLevel,
        allow_follow_up: record.allow_follow_up !== false,
      } satisfies GeneratedQuestion;

      const resolvedMappedSkill = resolveMappedSkill(normalizedQuestion, context);

      if (!resolvedMappedSkill) {
        return null;
      }

      return {
        ...normalizedQuestion,
        mapped_skill: resolvedMappedSkill,
      } satisfies GeneratedQuestion;
    })
    .filter((item): item is GeneratedQuestion => item !== null);
}

function isTooSimilar(
  questionText: string,
  sourceType: QuestionSource,
  acceptedQuestions: GeneratedQuestion[]
) {
  const newIntent = detectIntent(questionText);
  const newTokens = tokenizeQuestion(questionText);
  const newLeadKey = getQuestionLeadKey(questionText);

  return acceptedQuestions.some((existing) => {
    const existingIntent = detectIntent(existing.question_text);
    const score = similarityScore(questionText, existing.question_text);
    const existingLeadKey = getQuestionLeadKey(existing.question_text);
    const sharedLongTokens = newTokens.filter((token) =>
      tokenizeQuestion(existing.question_text).includes(token)
    );

    return (
      score >= 0.58 ||
      (newLeadKey !== "" &&
        newLeadKey === existingLeadKey &&
        existing.source_type === sourceType) ||
      (newIntent === existingIntent &&
        sharedLongTokens.length >= 3 &&
        existing.source_type === sourceType)
    );
  });
}

function buildReferenceContext(params: {
  sourceType: QuestionSource;
  roleTitle: string | null;
  anchor: string;
  intent: string;
  claim?: string | null;
}) {
  return {
    role_title: params.roleTitle,
    anchor: params.anchor,
    intent: params.intent,
    claim: params.claim ?? null,
  } satisfies Record<string, unknown>;
}

function buildResumeQuestion(params: {
  roleTitle: string | null;
  skillOrClaim: string;
  index: number;
}): GeneratedQuestion {
  const focus = normalizeText(params.skillOrClaim) || "your background";
  const cleanedFocus = cleanAnchorText(focus) || "your background";
  const templates = [
    `You highlighted ${cleanedFocus}. What was the most demanding piece of work around it, and how did you deliver the result?`,
    `Your resume points to ${cleanedFocus}. What problem forced you to go deepest technically, and what did you change?`,
    `When ${cleanedFocus} mattered most in your work, what outcome were you responsible for and how did you get there?`,
  ];

  return {
    question_text: templates[params.index % templates.length],
    source_type: "resume",
    reference_context: buildReferenceContext({
      sourceType: "resume",
      roleTitle: params.roleTitle,
      anchor: cleanedFocus,
      intent: detectIntent(templates[params.index % templates.length]),
      claim: cleanedFocus,
    }),
    mapped_skill: cleanedFocus,
    phase_hint: params.index === 0 ? "warmup" : "core",
    difficulty_level: 3,
    allow_follow_up: true,
  };
}

function buildJobQuestion(params: {
  roleTitle: string | null;
  skill: string;
  responsibility?: string | null;
  index: number;
}): GeneratedQuestion {
  const focus =
    cleanAnchorText(params.skill) ||
    cleanAnchorText(params.responsibility) ||
    "the core responsibilities";
  const templates = [
    `How do you troubleshoot ${focus} when it starts failing in production?`,
    `If you had to improve ${focus} under real delivery pressure, what would you examine first?`,
    `What signals tell you ${focus} is degrading, and what actions do you take next?`,
    `Walk me through how you would execute ${focus} reliably in a live environment.`,
  ];

  return {
    question_text: templates[params.index % templates.length],
    source_type: "job",
    reference_context: buildReferenceContext({
      sourceType: "job",
      roleTitle: params.roleTitle,
      anchor: focus,
      intent: detectIntent(templates[params.index % templates.length]),
    }),
    mapped_skill: focus,
    phase_hint: params.index >= 2 ? "probe" : "core",
    difficulty_level: 3 + (params.index % 2),
    allow_follow_up: true,
  };
}

function buildBehavioralQuestion(params: {
  roleTitle: string | null;
  skill: string;
  index: number;
}): GeneratedQuestion {
  const skill = normalizeText(params.skill) || "your work";
  const cleanedSkill = cleanAnchorText(skill) || "your work";
  const templates = [
    `When work around ${cleanedSkill} started going off track, how did you regain control and align the team?`,
    `Think of a time you had conflicting priorities around ${cleanedSkill}. How did you decide what to protect first?`,
    `When people pushed for speed on ${cleanedSkill}, how did you defend quality, reliability, or risk controls?`,
  ];

  return {
    question_text: templates[params.index % templates.length],
    source_type: "behavioral",
    reference_context: buildReferenceContext({
      sourceType: "behavioral",
      roleTitle: params.roleTitle,
      anchor: cleanedSkill,
      intent: detectIntent(templates[params.index % templates.length]),
    }),
    mapped_skill: cleanedSkill,
    phase_hint: "probe",
    difficulty_level: 3,
    allow_follow_up: true,
  };
}

function buildFallbackQuestion(
  sourceType: QuestionSource,
  context: InterviewContext,
  index: number
): GeneratedQuestion {
  if (sourceType === "resume") {
    const focus = pickQuestionAnchor({
      sourceType,
      overlapSkills: context.overlapSkills,
      jobSkills: [...context.job.coreSkills, ...context.interviewSkills],
      responsibilities: context.job.responsibilities,
      resumeSkills: context.resume.skills,
      resumeClaims: context.resume.claims,
      roleTitle: context.roleTitle,
      index,
    });

    return buildResumeQuestion({
      roleTitle: context.roleTitle,
      skillOrClaim: focus,
      index,
    });
  }

  if (sourceType === "job") {
    const focus = pickQuestionAnchor({
      sourceType,
      overlapSkills: context.overlapSkills,
      jobSkills: [...context.job.coreSkills, ...context.interviewSkills],
      responsibilities: context.job.responsibilities,
      resumeSkills: context.resume.skills,
      resumeClaims: context.resume.claims,
      roleTitle: context.roleTitle,
      index,
    });

    return buildJobQuestion({
      roleTitle: context.roleTitle,
      skill: focus,
      responsibility: context.job.responsibilities[index % Math.max(context.job.responsibilities.length, 1)] ?? null,
      index,
    });
  }

  const focus = pickQuestionAnchor({
    sourceType,
    overlapSkills: context.overlapSkills,
    jobSkills: [...context.job.coreSkills, ...context.interviewSkills],
    responsibilities: context.job.responsibilities,
    resumeSkills: context.resume.skills,
    resumeClaims: context.resume.claims,
    roleTitle: context.roleTitle,
    index,
  });

  return buildBehavioralQuestion({
    roleTitle: context.roleTitle,
    skill: focus,
    index,
  });
}

function distributeQuestions(
  generatedQuestions: GeneratedQuestion[],
  distribution: Distribution,
  context: InterviewContext
) {
  const accepted: GeneratedQuestion[] = [];
  const acceptedBySource: Record<QuestionSource, GeneratedQuestion[]> = {
    resume: [],
    job: [],
    behavioral: [],
  };

  for (const question of generatedQuestions) {
    if (isGenericQuestion(question.question_text)) {
      continue;
    }

    if (hasWeakLeadIn(question.question_text)) {
      continue;
    }

    if (containsLocationNoise(question.question_text) || containsGenderAssumption(question.question_text)) {
      continue;
    }

    if (!isReadableQuestion(question.question_text)) {
      continue;
    }

    if (!isContextualQuestion(question, context)) {
      continue;
    }

    if (!isSkillDrivenQuestion(question, context)) {
      continue;
    }

    if (isTooSimilar(question.question_text, question.source_type, accepted)) {
      continue;
    }

    if (acceptedBySource[question.source_type].length >= distribution[question.source_type]) {
      continue;
    }

    accepted.push(question);
    acceptedBySource[question.source_type].push(question);
  }

  for (const sourceType of ["job", "resume", "behavioral"] as QuestionSource[]) {
    let index = 0;

    while (acceptedBySource[sourceType].length < distribution[sourceType]) {
      const fallback = buildFallbackQuestion(sourceType, context, index);

      if (
        !isGenericQuestion(fallback.question_text) &&
        !hasWeakLeadIn(fallback.question_text) &&
        !containsLocationNoise(fallback.question_text) &&
        !containsGenderAssumption(fallback.question_text) &&
        isReadableQuestion(fallback.question_text) &&
        isSkillDrivenQuestion(fallback, context) &&
        !isTooSimilar(fallback.question_text, fallback.source_type, accepted)
      ) {
        accepted.push(fallback);
        acceptedBySource[sourceType].push(fallback);
      }

      index += 1;

      if (index > 20) {
        break;
      }
    }
  }

  const orderedSources: QuestionSource[] = [];
  const remaining = { ...distribution };

  while (orderedSources.length < Object.values(distribution).reduce((sum, value) => sum + value, 0)) {
    if (remaining.resume > 0) {
      orderedSources.push("resume");
      remaining.resume -= 1;
    }

    if (remaining.job > 0) {
      orderedSources.push("job");
      remaining.job -= 1;
    }

    if (remaining.job > 0) {
      orderedSources.push("job");
      remaining.job -= 1;
    }

    if (remaining.behavioral > 0) {
      orderedSources.push("behavioral");
      remaining.behavioral -= 1;
    }
  }

  const sourceQueues: Record<QuestionSource, GeneratedQuestion[]> = {
    resume: [...acceptedBySource.resume],
    job: [...acceptedBySource.job],
    behavioral: [...acceptedBySource.behavioral],
  };

  return orderedSources
    .map((sourceType) => sourceQueues[sourceType].shift() ?? null)
    .filter((question): question is GeneratedQuestion => question !== null);
}

Deno.serve(async (request) => {
  if (request.method === "OPTIONS") {
    return new Response("ok", { headers: CORS_HEADERS });
  }

  if (request.method !== "POST") {
    return jsonResponse({ success: false, error: "Method not allowed" }, 405);
  }

  try {
    const { interview_id: interviewId } = (await request.json()) as {
      interview_id?: string;
    };

    if (!interviewId?.trim()) {
      return jsonResponse({ success: false, error: "interview_id is required" }, 400);
    }

    const supabase = createSupabase();

    const { data: interview, error: interviewError } = await supabase
      .from("interviews")
      .select("interview_id, candidate_id, job_id, duration_minutes, question_count, ai_strictness")
      .eq("interview_id", interviewId)
      .single();

    if (interviewError || !interview) {
      return jsonResponse({ success: false, error: "Interview not found" }, 404);
    }

    const [{ data: resume }, { data: job }, { data: templateConfig }, { data: skills }] =
      await Promise.all([
        supabase
          .from("candidate_resume_ai")
          .select("raw_resume, extracted_skills, extracted_claims, claimed_experience_years")
          .eq("interview_id", interviewId)
          .order("created_at", { ascending: false })
          .limit(1)
          .maybeSingle(),
        supabase
          .from("job_positions")
          .select("job_title, job_description, core_skills, difficulty_profile")
          .eq("job_id", interview.job_id)
          .single(),
        supabase
          .from("interview_configs")
          .select("template_id, total_duration_minutes, mode")
          .eq("interview_id", interviewId)
          .maybeSingle(),
        supabase
          .from("interview_skill_map")
          .select("skill_id, skill_master(skill_name, skill_code)")
          .eq("interview_id", interviewId),
      ]);

    const context: InterviewContext = {
      roleTitle: cleanRoleTitle(job?.job_title) ?? null,
      resume: {
        rawResume: resume?.raw_resume ?? null,
        skills: uniqueStrings(asStringArray(resume?.extracted_skills)).slice(0, 12),
        claims: extractClaimPhrases(resume?.extracted_claims),
        experienceYears:
          typeof resume?.claimed_experience_years === "number"
            ? resume.claimed_experience_years
            : null,
      },
      job: {
        title: cleanRoleTitle(job?.job_title) ?? null,
        description: normalizeText(job?.job_description) || null,
        coreSkills: uniqueStrings(asStringArray(job?.core_skills)).slice(0, 12),
        responsibilities: extractResponsibilityAnchors(job?.job_description),
        difficultyProfile: normalizeText(job?.difficulty_profile) || null,
      },
      interviewSkills: uniqueStrings(
        (skills ?? []).flatMap((item) => {
          const skill = item?.skill_master as
            | { skill_name?: string | null; skill_code?: string | null }
            | null
            | undefined;

          return [skill?.skill_name ?? null, skill?.skill_code ?? null];
        })
      ).slice(0, 12),
      overlapSkills: [],
    };
    context.overlapSkills = computeSkillOverlap({
      resumeSkills: context.resume.skills,
      resumeClaims: context.resume.claims,
      jobSkills: [...context.job.coreSkills, ...context.interviewSkills],
    });

    const targetQuestionCount = resolveEffectiveQuestionCount({
      configuredCount: interview.question_count,
      durationMinutes:
        typeof interview.duration_minutes === "number"
          ? interview.duration_minutes
          : templateConfig?.total_duration_minutes,
    });
    const distribution = buildDistribution(
      targetQuestionCount,
      context.resume.skills.length > 0 || context.resume.claims.length > 0
    );

    const prompt = await createJsonChatCompletion(
      [
        {
          role: "system",
          content: [
            "You are generating role-specific interview questions.",
            `Return exactly ${targetQuestionCount} questions in JSON under key "questions".`,
            `Question distribution must be: ${distribution.resume} resume, ${distribution.job} job, ${distribution.behavioral} behavioral.`,
            "Strict rules:",
            "- Do not ask generic questions such as 'tell me about yourself' or 'tell me about your experience'.",
            "- Prefer job core skills and job responsibilities over job title when phrasing questions.",
            "- First look for overlap between resume skills/claims and job-required skills. Use that overlap whenever possible.",
            "- If there is no overlap, use job-required skills before any other context.",
            "- Every question must be anchored to candidate skills, job-required skills, or both.",
            "- mapped_skill must be a real skill from resume skills, job-required skills, or their overlap. Do not invent a skill label.",
            "- Do not use job title as the primary anchor for any question.",
            "- Never include location words like Delhi, Bangalore, Remote, or shift information in the final question.",
            "- Never use he/she/his/her. Use neutral language only.",
            "- Use a diverse mix of scenario-based, tool-based, problem-solving, and real-world execution questions.",
            "- Avoid repetitive lead-ins like 'describe a situation', 'share an example', and 'what trade-offs' unless there is no better phrasing.",
            "- Every resume question must reference a resume skill, resume claim, or claimed experience.",
            "- Every job question must reference a required skill or a job responsibility. Use job title only as a last-resort fallback.",
            "- Every behavioral question must still be anchored to a relevant skill or responsibility.",
            "- Do not repeat the same intent with different wording.",
            "- Vary intent across optimization, troubleshooting, design, migration, ownership, decision-making, and trade-offs.",
            "- Each question must be concise, specific, and interviewer-style.",
            "- Each question must include mapped_skill when possible.",
            "- Each question must include source_type, reference_context, phase_hint, difficulty_level, and allow_follow_up.",
            "Return only JSON.",
          ].join("\n"),
        },
        {
          role: "user",
          content: JSON.stringify(
            {
              interview: {
                ...interview,
                target_question_count: targetQuestionCount,
                target_distribution: distribution,
              },
              resume: {
                raw_resume: context.resume.rawResume,
                extracted_skills: context.resume.skills,
                extracted_claims: context.resume.claims,
                claimed_experience_years: context.resume.experienceYears,
              },
              job: {
                job_title: context.job.title,
                job_description: context.job.description,
                core_skills: context.job.coreSkills,
                responsibilities: context.job.responsibilities,
                difficulty_profile: context.job.difficultyProfile,
              },
              interview_skills: context.interviewSkills,
              overlap_skills: context.overlapSkills,
              generation_rules: {
                prioritize_overlap_first: true,
                reject_generic: true,
                reject_repetition: true,
                priority_order: [
                  "resume-job skill overlap",
                  "uncovered job skills",
                  "unverified resume claims",
                  "behavioral fallback",
                ],
              },
            },
            null,
            2
          ),
        },
      ],
      0.25
    );

    const normalizedQuestions = normalizeQuestions(prompt, context);
    const questions = distributeQuestions(normalizedQuestions, distribution, context);

    if (questions.length === 0) {
      throw new Error("No valid questions generated after validation");
    }

    await supabase
      .from("interview_questions")
      .delete()
      .eq("interview_id", interviewId)
      .eq("is_dynamic", true);

    const skillNames = uniqueStrings([
      ...questions.map((question) => question.mapped_skill ?? null),
      ...context.job.coreSkills,
      ...context.interviewSkills,
    ]);

    const { data: skillRows } = skillNames.length
      ? await supabase
          .from("skill_master")
          .select("skill_id, skill_name, skill_code")
          .or(
            skillNames
              .map((skill) => `skill_name.eq.${skill},skill_code.eq.${skill}`)
              .join(",")
          )
      : { data: [] };

    const skillMap = new Map<string, string>();
    for (const row of skillRows ?? []) {
      if (typeof row.skill_name === "string") {
        skillMap.set(row.skill_name.toLowerCase(), row.skill_id);
      }
      if (typeof row.skill_code === "string") {
        skillMap.set(row.skill_code.toLowerCase(), row.skill_id);
      }
    }

    const insertRows = questions.map((question, index) => ({
      interview_id: interviewId,
      question_id: null,
      question_order: index + 1,
      is_mandatory: true,
      allow_follow_up: question.allow_follow_up ?? true,
      question_text: question.question_text,
      question_type:
        question.source_type === "behavioral" ? "behavioral" : "open_ended",
      source_type: question.source_type,
      reference_context: {
        ...question.reference_context,
        distribution_slot: question.source_type,
        memory_key: normalizeQuestionText(question.question_text),
      },
      is_dynamic: true,
      phase_hint: question.phase_hint ?? "core",
      difficulty_level: question.difficulty_level ?? 3,
      target_skill_id: question.mapped_skill
        ? skillMap.get(question.mapped_skill.toLowerCase()) ?? null
        : null,
    }));

    const { error: insertError } = await supabase
      .from("interview_questions")
      .insert(insertRows);

    if (insertError) {
      throw new Error(`Failed to insert interview questions: ${insertError.message}`);
    }

    return jsonResponse({
      success: true,
      questions_inserted: insertRows.length,
      distribution,
      resume_stored: Boolean(context.resume.rawResume),
    });
  } catch (error) {
    console.error("[generate-questions]", error);
    return jsonResponse(
      {
        success: false,
        error: error instanceof Error ? error.message : "Unexpected error",
      },
      500
    );
  }
});
